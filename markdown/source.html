# Tests for interpret_directives
test_cases = [
    {
        'name': 'No directives',
        'directives': {},
        'expected_should_echo': True,
        'expected_should_show_output': True,
    },
    {
        'name': 'Echo false',
        'directives': {'echo': False},
        'expected_should_echo': False,
        'expected_should_show_output': True,
    },
    {
        'name': 'Echo true',
        'directives': {'echo': True},
        'expected_should_echo': True,
        'expected_should_show_output': True,
    },
    {
        'name': 'Output false',
        'directives': {'output': False},
        'expected_should_echo': True,
        'expected_should_show_output': False,
    },
    {
        'name': 'Output true',
        'directives': {'output': True},
        'expected_should_echo': True,
        'expected_should_show_output': True,
    },
    {
        'name': 'Hide only',
        'directives': {'hide': None},
        'expected_should_echo': False,
        'expected_should_show_output': False,
    },
    # There are other weird cases like hide and echo: true, but these
    # should not happen in practice and I don't want to have tests for
    # the arbitrary behavior that would happen.
]

for tc in test_cases:
    name = tc['name']
    directives = tc['directives']
    expected_should_echo = tc['expected_should_echo']
    expected_should_show_output = tc['expected_should_show_output']

    print(f"Running test case: {name}")
    should_echo, should_show_output = interpret_directives(directives)

    test_eq(should_echo, expected_should_echo)
    test_eq(should_show_output, expected_should_show_output)

MarkdownSourceHandler.handle_markdown

 MarkdownSourceHandler.handle_markdown (lines:Sequence[str])

Handle the source lines for a markdown cell

Type Details
lines typing.Sequence[str] The lines of source text 
markdown_source = [
    '# This is Markdown\n',
    'This is a [link](https://www.google.com).\n',
    'This is:\n',
    '\n',
    '* a \n',
    '* bulleted\n',
    '* list\n',
    '\n',
    'Yup.',
]
stream = io.StringIO()
src_handler = MarkdownSourceHandler(stream)

src_handler.handle_markdown(markdown_source)

stream.seek(0)
output = stream.read()

expected = """\
# This is Markdown
This is a [link](https://www.google.com).
This is:

* a""" + " " + """
* bulleted
* list

Yup.

"""

test_eq(output, expected)
print(output)
# This is Markdown
This is a [link](https://www.google.com).
This is:

* a 
* bulleted
* list

Yup.

MarkdownSourceHandler.handle_python_source

 MarkdownSourceHandler.handle_python_source (lines:Sequence[str])

Handle the source lines for a python code cell

Type Details
lines typing.Sequence[str] The lines of source text
Returns bool Indicates whether the output should be handled for this cell. 
python_source = ['name = \'world\'\n', 'print(f"hello, {name}")']

stream = io.StringIO()
src_handler = MarkdownSourceHandler(stream)

should_show_output = src_handler.handle_python_source(python_source)

stream.seek(0)
output = stream.read()

expected = """\
```python
name = 'world'
print(f"hello, {name}")
```

"""

test_eq(output, expected)
test_eq(should_show_output, True)
print(output)

print(f"hello, {name}")
